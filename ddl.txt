
CREATE DATABASE gravy;

title VARCHAR(100) NOT NULL,
url VARCHAR(250),
view_count VARCHAR(100),

title VARCHAR(100) NOT NULL,
url VARCHAR(250),
view_count VARCHAR(100),
subscribers VARCHAR(100),
post_date VARCHAR(100),
created_at timestamp DEFAULT clock_timestamp()
);
